    # Optimized for expressibility and trainability while maintaining reliability
    n_layers = 2

    theta = ParameterVector('θ', length=total_params)
    
    qc = QuantumCircuit(num_qubits)
    
    # Layer structure: U3 rotations + optimized entangling
    for layer in range(n_layers):
        # U3 rotation layer

            qc.ry(theta[idx], i)
            idx += num_qubits
            qc.rz(theta[idx], i)
            idx += num_qubits
            qc.ry(theta[idx], i)

        if layer < n_layers - 1:

                qc.cz(i, i + 1)
            for i in range(1, num_qubits - 1, 2):
                qc.cz(i, i + 1)
    
    return qc, list(theta)
