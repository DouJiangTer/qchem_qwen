                for k in range(j, i):
                    qc.cx(k, k+1)
                
                # Apply rotation on target
                qc.ry(params[idx], i)
                
                # Controlled rotation
                qc.cx(j, i)
                qc.ry(-params[idx]/2, i)
                qc.cx(j, i)
                qc.ry(params[idx]/2, i)
                
                # Reverse ladder
                for k in range(i-1, j, -1):
                    qc.cx(k, k+1)
    
    # Apply double excitations using 12-CNOT decomposition
    start_idx = len(singles)
    for idx, (i, j, k, l) in enumerate(doubles):
        param_idx = start_idx + idx
        
        # Sort qubits in descending order for standard decomposition
        qubits = sorted([i, j, k, l], reverse=True)
        i, j, k, l = qubits
        
        # Standard double excitation decomposition (12 CNOTs)
        # First CNOT ladder to bring qubits together
        qc.cx(l, k)
        qc.cx(k, j)
        qc.cx(j, i)
        
        # Apply controlled rotations
        qc.ry(params[param_idx], i)
        qc.cx(i, j)
        qc.ry(-params[param_idx]/2, j)
        qc.cx(i, j)
        qc.ry(params[param_idx]/2, j)
        
        qc.cx(j, k)
        qc.ry(-params[param_idx]/2, k)
        qc.cx(j, k)
        qc.ry(params[param_idx]/2, k)
        
        qc.cx(k, l)
        qc.ry(-params[param_idx]/2, l)
        qc.cx(k, l)
        qc.ry(params[param_idx]/2, l)
        
        # Reverse CNOT ladder
        qc.cx(j, i)
        qc.cx(k, j)
        qc.cx(l, k)
    
    return qc, params
