    # Virtual to virtual excitations (4->5, 6->7)
    circuit.cx(4, 5)
    circuit.ry(theta_single[6], 5)
    circuit.cx(4, 5)
    
    circuit.cx(6, 7)
    circuit.ry(theta_single[7], 7)
    circuit.cx(6, 7)
    
    # Implement double excitation using the standard 4-qubit circuit
    # Double excitation between spin orbitals 0,1 and 2,3
    # Using the circuit from arXiv:1805.03469
    qubits = [0, 1, 2, 3]
    
    # Apply Hadamard gates
    for q in qubits:
        circuit.h(q)
    
    # Apply CNOTs
    circuit.cx(qubits[0], qubits[2])
    circuit.cx(qubits[1], qubits[3])
    
    # Apply controlled rotation
    circuit.cry(2 * theta_double[0], qubits[2], qubits[3])
    
    # Reverse CNOTs
    circuit.cx(qubits[1], qubits[3])
    circuit.cx(qubits[0], qubits[2])
    
    # Apply Hadamard gates
    for q in qubits:
        circuit.h(q)
    
    # Combine all parameters
    parameters = list(theta_single) + list(theta_double)
    
    return circuit, parameters
