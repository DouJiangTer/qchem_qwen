def build_ansatz(num_qubits):
    # H3+ in STO-3G basis with 6 qubits
    # Use a chemically inspired ansatz: UCCSD-like with single and double excitations
    # Adapted for the specific molecular system and qubit count
    
    qc = QuantumCircuit(num_qubits)
    params = ParameterVector('θ', length=8)
    
    # Reference: Hartree-Fock state for H3+ (2 electrons, so 2 orbitals occupied in minimal basis)
    # In 6-qubit representation after Jordan-Wigner: HF state is |110000⟩
    qc.x(0)
    qc.x(1)
    
    # Single excitations (i -> a, where i is occupied, a is virtual)
    # For H3+ STO-3G: occupied orbitals 0,1; virtual orbitals 2,3,4,5
    
    # Single excitation 0->2
    qc.cry(params[0], 0, 2)
    qc.cx(0, 2)
    qc.cry(-params[0], 0, 2)
    
    # Single excitation 0->3
    qc.cry(params[1], 0, 3)
    qc.cx(0, 3)
    qc.cry(-params[1], 0, 3)
    
    # Single excitation 1->4
    qc.cry(params[2], 1, 4)
    qc.cx(1, 4)
    qc.cry(-params[2], 1, 4)
    
    # Single excitation 1->5
    qc.cry(params[3], 1, 5)
    qc.cx(1, 5)
    qc.cry(-params[3], 1, 5)
    
    # Double excitation 0,1 -> 2,3 (most important for H3+)
    # Using standard decomposition for double excitation gate
    qc.cry(params[4]/2, 1, 3)
    qc.cx(1, 3)
    qc.cry(params[4]/2, 0, 3)
    qc.cx(0, 3)
    qc.cry(-params[4]/2, 1, 3)
    qc.cx(1, 3)
    qc.cry(-params[4]/2, 0, 3)
    qc.cx(0, 3)

    # Additional single excitation 0->5
    qc.cry(params[7], 0, 5)
    qc.cx(0, 5)
    qc.cry(-params[7], 0, 5)
    
    return qc, params
