from qiskit import QuantumCircuit

def build_ansatz(num_qubits):
    # Create parameter vector for trainable rotation angles
    params = ParameterVector('θ', length=16)
    
    # Initialize quantum circuit
    qc = QuantumCircuit(num_qubits)
    
    # Prepare Hartree-Fock state (for H2 in 6-31G with 8 qubits, HF state is |11000000⟩)
    qc.x(0)
    qc.x(1)
    
    # Add first layer of Y rotations
    for i in range(num_qubits):
        qc.ry(params[i], i)
    
    # Add entangling CZ gates in a ladder pattern
    for i in range(num_qubits - 1):
        qc.cz(i, i + 1)
    
    # Add second layer of Y rotations
    for i in range(num_qubits):
        qc.ry(params[i + num_qubits], i)
    
    # Add entangling CZ gates in a ladder pattern again
    for i in range(num_qubits - 1):
        qc.cz(i, i + 1)
    
    return qc, params
