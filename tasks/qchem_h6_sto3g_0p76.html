    # Layer 3: Entangling block with nearest-neighbor CNOTs in linear topology
    # Optimized for transpilation efficiency
    for i in range(num_qubits - 1):
        qc.cx(i, i + 1)
    
    # Layer 4: Additional parameterized rotations to improve expressibility
    # Use different parameter subset to avoid redundant operations
    for i in range(num_qubits):
        qc.ry(rot_params[3*i + 1], i)
        qc.rz(rot_params[3*i], i)
        qc.ry(rot_params[3*i + 1], i)
    
    return qc, params
