                    qc.cx(i, a)

                    qc.cx(i, a)
                    param_idx_s += 1
    
    # Double excitations: i,j -> a,b
    # Only include physically relevant pairs (conserving spin and symmetry)
    # Simplified set of double excitations
    double_pairs = [
