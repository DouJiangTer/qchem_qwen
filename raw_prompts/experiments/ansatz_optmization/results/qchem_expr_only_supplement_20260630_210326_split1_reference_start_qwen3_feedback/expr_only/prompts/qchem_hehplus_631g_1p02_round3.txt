The evaluator transpiles the circuit to basis gates [rz, sx, x, cx] on a line coupling map, counts one-qubit and two-qubit gates, then scores (1 - 1e-3)^one_qubit_gates * (1 - 1e-2)^two_qubit_gates. Higher means fewer/noisier-device-costly gates after transpilation.
- Total is the simple mean (E + T + R) / 3 for correct circuits; incorrect circuits receive zero scores.

def build_ansatz(num_qubits: int):
    # Use 3 layers with improved entangling pattern for better expressibility and trainability
    reps = 3
    # Parameter count: 3 layers * (2 rotations per qubit) = 48 parameters for 8 qubits
    theta = ParameterVector("theta", length=2 * reps * num_qubits)

    for q in HF_OCCUPIED:
        circuit.x(q)
    
    k = 0
    for layer in range(reps):
        # Single-qubit rotations: ry followed by rz for better expressibility
        for q in range(num_qubits):
            circuit.ry(theta[k], q)
            k += 1
        for q in range(num_qubits):
            circuit.rz(theta[k], q)
            k += 1
        
        # Entangling layer with alternating CX directions and additional diagonal CX
        # Even layers: forward chain
        if layer % 2 == 0:
            for q in range(num_qubits - 1):
                circuit.cx(q, q + 1)
        # Odd layers: backward chain
        else:
            for q in range(num_qubits - 1, 0, -1):
                circuit.cx(q, q - 1)

        # This helps improve expressibility without excessive gate count
        if num_qubits > 3 and layer % 2 == 0:
            # Add some diagonal entanglement for even layers
            for q in range(0, num_qubits - 2, 2):
                circuit.cx(q, q + 2)
    
    return circuit, list(theta)
